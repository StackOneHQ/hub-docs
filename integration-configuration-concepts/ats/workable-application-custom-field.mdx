## Create a Workable Application Custom Field

<Steps>
  <Step title="Click on your profile icon">
    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Click on the profile icon"
        src="/images/workable/custom-field/custom-field-01.png"
      />
    </Frame>
  </Step>
  <Step title="Navigate to Settings">
    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Navigate to settings"
        src="/images/workable/custom-field/custom-field-02.png"
      />
    </Frame>
  </Step>
  <Step title="Click on Workflow">
    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-03.png"
      />
    </Frame>
  </Step>
  <Step title="Scroll to the Candidate Custom Fields section">
    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-04.png"
      />
    </Frame>
  </Step>
  <Step title="Fields sections">
    Find or Create a section where you want the custom field created.

    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-05.png"
      />
    </Frame>
  </Step>
  <Step title="Add a new Custom Field">
    Hover over the desired section and click on the `+` icon to add a new custom field

    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-06.png"
      />
    </Frame>
  </Step>
  <Step title="Custom field form">
    1. Enter a `Custom field title`
    2. Select the appropriate  `Field type` 
    3. Set the `Provider` to `User` 
    4. Click `Add Custom field`

    <Warning>
      The Provider field must be set to User to be able to edit the field via API
    </Warning>
    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-08.png"
      />
    </Frame>
  </Step>
  <Step title="Save the changes">
    Scroll to the bottom and click <b>
    
    Save Changes</b>

    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-09.png"
      />
    </Frame>
  </Step>
  <Step title="Created custom field">
    Here is the <b>
    
    success </b>

    message shown after the custom field is created

    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-10.png"
      />
    </Frame>
  </Step>
  <Step title="Application form preview">
    You can preview where this sits on the candiate profile by Clicking the `Preview` button.

    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-11.png"
      />
    </Frame>
  </Step>
  <Step title="Preview candidate custom fields">
    Ensure you Select `Candidate profile` 

    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-12.png"
      />
    </Frame>
    Scroll down to see the newly created custom field

    <Frame>
      <img
        className="rounded-md"
        style={{ margin:"0 auto",border:"1px solid #efefef" }}
        alt="Search Create Custom Object"
        src="/images/workable/custom-field/custom-field-13.png"
      />
    </Frame>
  </Step>
</Steps>
