</Warning>
 
## Create Workable Application Custom Field
 
<Steps>
  <Step title="Click on Profile Icon">
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Profile icon location in Workable" src="/images/workable/custom-field/custom-field-01.png" />
    </Frame>
  </Step>
  <Step title="Click on Settings">
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Settings menu option in dropdown" src="/images/workable/custom-field/custom-field-02.png" />
    </Frame>
  </Step>
  <Step title="Click on Workflow">
   Click on <b>`Workflow`</b> and Scroll down to the bottom of the page and go to the <b>Candidate Custom Fields</b> section.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Workflow option" src="/images/workable/custom-field/custom-field-03.png" />
    </Frame>
  </Step>
  <Step title="Candidate Custom Fields Section">
    Here is the section where you can add custom fields.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Candidate custom fields section location" src="/images/workable/custom-field/custom-field-04.png" />
    </Frame>
  </Step>
  <Step title="Fields Section">
    Here are different groups of custom fields.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Groups of custom fields in Workable" src="/images/workable/custom-field/custom-field-05.png" />
    </Frame>
  </Step>
  <Step title="Add a Custom Field">
    Hover over the desired section and click on the <b>+</b> icon to add a new custom field.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="create custom field + icon" src="/images/workable/custom-field/custom-field-06.png" />
    </Frame>
  </Step>
  <Step title="Custom Field Form">
    Here is the form to create a custom field.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="custom field creation form" src="/images/workable/custom-field/custom-field-07.png" />
    </Frame>
  </Step>
  <Step title="Complete the Custom Field Form">
    Enter a <b>custom field title</b>, choose a <b>Field Type</b>, and set <b>Provider = User</b> from the dropdown menu.
    (The key requirement is that the custom field MUST be set to Provider = User.)
 
    Click <b>Add Custom Field</b> and scroll to the bottom.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Updated custom field form " src="/images/workable/custom-field/custom-field-08.png" />
    </Frame>
  </Step>
  <Step title="Application Form Section">
    Click the <b>Save Changes</b> button.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Save Changes button location" src="/images/workable/custom-field/custom-field-09.png" />
    </Frame>
  </Step>
  <Step title="Created Custom Field">
    Here is the <b>success</b> message confirming that the custom field was created.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Success message" src="/images/workable/custom-field/custom-field-10.png" />
    </Frame>
  </Step>
  <Step title="Application Form Preview">
    Click on <b>Preview</b> to see the custom fields.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Preview option for application form" src="/images/workable/custom-field/custom-field-11.png" />
    </Frame>
  </Step>
  <Step title="Preview Candidate Custom Fields">
    Click on a candidate profile and scroll down.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Candidate profile view" src="/images/workable/custom-field/custom-field-12.png" />
    </Frame>
    You will see the newly created custom field.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Newly created custom field in candidate profile" src="/images/workable/custom-field/custom-field-13.png" />
    </Frame>
  </Step>
  <Step title="Verify Custom Field Location">
    You can also see the custom field in the <b>Candidate Profile</b> inside the <b>Candidate Custom Fields Section</b> in the group where you created it.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Custom field location in candidate custom fields section" src="/images/workable/custom-field/custom-field-14.png" />
    </Frame>
  </Step>
</Steps>
