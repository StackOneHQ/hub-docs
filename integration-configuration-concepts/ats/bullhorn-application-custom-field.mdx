  This guidance assumes that you have admin privileges for your BUllhorn account.
</Warning>

## Create Bullhorn Application Custom Field
<Steps>
  <Step title="Click on Menu">
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Dashboard window" src="/images/bullhorn/custom-field/custom-field-01.png" />
    </Frame>
  </Step>

  <Step title="Click on Candidates">
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Admin menu" src="/images/bullhorn/custom-field/custom-field-02.png" />
    </Frame>
  </Step>

  <Step title="Click on Columns">
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Candidate table" src="/images/bullhorn/custom-field/custom-field-03.png" />
    </Frame>
  </Step>

  <Step title="Custom Fields Table">
    Select the desired <b>`custom fields`</b> from the dropdown menu.
    Then, click the <b>`Save`</b> button.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Custom fields column" src="/images/bullhorn/custom-field/custom-field-04.png" />
    </Frame>
  </Step>

  <Step title="Candidates">
    The <b>Candidate Table</b> now includes the custom field. You can move it to the desired position by dragging and dropping it.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Custom fields view in the table" src="/images/bullhorn/custom-field/custom-field-05.png" />
    </Frame>
  </Step>

  <Step title="Candidates Table">
    Here is the candidate table. You can click on any candidate to view their details.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Candidate profile" src="/images/bullhorn/custom-field/custom-field-06.png" />
    </Frame>
  </Step>

  <Step title="Candidate Profile">
    Here is the candidate profile. Click on <b>`Edit`</b> to view and add a custom field value.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Candidate profile with multiple fields" src="/images/bullhorn/custom-field/custom-field-07.png" />
    </Frame>
  </Step>

  <Step title="Candidate Profile Edit Section">
    In the candidate profile edit section, scroll down to find the custom field.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Candidate profile edit section" src="/images/bullhorn/custom-field/custom-field-08.png" />
    </Frame>
  </Step>

  <Step title="Custom Field">
    Here is the candidate profile edit section where you can view the custom field.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Custom field" src="/images/bullhorn/custom-field/custom-field-09.png" />
    </Frame>
  </Step>
</Steps>

## Configure Bullhorn Custom Field

<Steps>
  <Step title="Click on Menu">
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Dashboard window" src="/images/bullhorn/custom-field/custom-field-01.png" />
    </Frame>
  </Step>

  <Step title="Click on Field Mappings">
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Admin menu" src="/images/bullhorn/custom-field/custom-field-config-01.png" />
    </Frame>
  </Step>

  <Step title="Click on Related Group">
    Scroll down and select any group to configure the custom field.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Fields Group" src="/images/bullhorn/custom-field/custom-field-config-02.png" />
    </Frame>
  </Step>

  <Step title="Job Submission">
    Here are the job submission-related fields. You can select any and set the configurations.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Fields Group" src="/images/bullhorn/custom-field/custom-field-config-03.png" />
    </Frame>
  </Step>

  <Step title="Custom Fields Configuration">
    Here is the custom fields configuration form. Set the configurations as per your requirements and click the <b>`Save`</b> button.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Fields Group" src="/images/bullhorn/custom-field/custom-field-config-04.png" />
    </Frame>
  </Step>

  <Step title="Field Maps">
    Scroll up and click the <b>`Save`</b> button to apply the changes.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Fields Group" src="/images/bullhorn/custom-field/custom-field-config-05.png" />
    </Frame>
  </Step>

  <Step title="Success Message">
    Here is the success message after saving the configuration.
    <Frame>
      <img className='rounded-md' style={{ margin: '0 auto', border: '1px solid #efefef' }} alt="Fields Group" src="/images/bullhorn/custom-field/custom-field-config-06.png" />
    </Frame>
  </Step>
</Steps>
